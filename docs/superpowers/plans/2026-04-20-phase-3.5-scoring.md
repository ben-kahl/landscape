```python
def score_candidate(
    vector_sim: float,
    graph_distance: int,
    edge_confidence: float,
    reinforcement: float,
    weights: ScoringWeights,
) -> float:
    """Combine the four signals into a final score under multiplicative gating.

    Base relevance is additive over vec_sim, proximity, and edge_confidence.
    Reinforcement acts as a bounded multiplier: score = base * (1 + γ·reinforcement).
    Max possible: (α + β + δ) · (1 + γ·cap)."""
    proximity = 1.0 / (1.0 + max(0, graph_distance))
    base = (
        weights.alpha * max(0.0, min(1.0, vector_sim))
        + weights.beta * proximity
        + weights.delta * max(0.0, min(1.0, edge_confidence))
    )
    return base * (1.0 + weights.gamma * reinforcement)


def max_possible_score(weights: ScoringWeights) -> float:
    """The theoretical ceiling under multiplicative gating. Used as an invariant
    in rumination tests."""
    base = weights.alpha + weights.beta + weights.delta
    return base * (1.0 + weights.gamma * weights.reinforcement_cap)
```

```python
@pytest.mark.asyncio
async def test_reasserted_fact_outranks_cold_fact(neo4j_driver):
    """Two facts seeded identically in the graph. Re-assert one three extra
    times via upsert_relation. A query that retrieves both should rank the
    reinforced one strictly higher.

    Uses upsert_* directly (not HTTP ingest) to isolate the reinforcement
    signal from extraction variability."""
    from datetime import UTC, datetime
    from landscape.retrieval.query import retrieve
    from landscape.storage import neo4j_store

    # Clean any prior state for these probe entities
    names = ["Reinforce Subject", "Reinforce Object Warm", "Reinforce Object Cold"]
    async with neo4j_driver.session() as session:
        await session.run(
            "MATCH (e:Entity) WHERE e.name IN $names DETACH DELETE e",
            names=names,
        )

    now = datetime.now(UTC).isoformat()
    # Seed three entities
    for name in names:
        await neo4j_store.upsert_entity(
            name=name,
            entity_type="Concept",
            source_doc="phase-3.5-test",
            confidence=0.9,
            model="test",
            created_by="ingest",
        )

    # Two relations: one warm (asserted 4x total), one cold (asserted 1x)
    await neo4j_store.upsert_relation(
        subject_name="Reinforce Subject",
        relation_type="RELATED_TO",
        object_name="Reinforce Object Cold",
        confidence=0.9,
        source_doc="phase-3.5-test",
        model="test",
        created_by="ingest",
    )
    for _ in range(4):
        await neo4j_store.upsert_relation(
            subject_name="Reinforce Subject",
            relation_type="RELATED_TO",
            object_name="Reinforce Object Warm",
            confidence=0.9,
            source_doc="phase-3.5-test",
            model="test",
            created_by="ingest",
        )

    # Also reinforce the warm object entity directly (4 extra touches)
    for _ in range(4):
        await neo4j_store.upsert_entity(
            name="Reinforce Object Warm",
            entity_type="Concept",
            source_doc="phase-3.5-test",
            confidence=0.9,
            model="test",
            created_by="ingest",
        )

    # Query close to the seeded names so both objects land in seeds
    result = await retrieve(
        query_text="Reinforce Object",
        hops=1,
        limit=10,
        reinforce=False,
    )
    names_in_order = [r.name for r in result.results]
    assert "Reinforce Object Warm" in names_in_order
    assert "Reinforce Object Cold" in names_in_order
    warm_idx = names_in_order.index("Reinforce Object Warm")
    cold_idx = names_in_order.index("Reinforce Object Cold")
    assert warm_idx < cold_idx, (
        f"Expected warm (access_count=5) to rank above cold (access_count=1); "
        f"got order {names_in_order}"
    )

    # Cleanup
    async with neo4j_driver.session() as session:
        await session.run(
            "MATCH (e:Entity) WHERE e.name IN $names DETACH DELETE e",
            names=names,
        )
```
